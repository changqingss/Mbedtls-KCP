cmake_minimum_required(VERSION 3.10)

project(kcp C)

# 添加动态库
add_library(
        kcp                # 库名
        SHARED             # 生成动态库
        ikcp.c             # 源文件
)

# 设置包含目录
target_include_directories(
        kcp PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}
)

# 可选优化
target_compile_options(kcp PRIVATE -O2 -fPIC)
